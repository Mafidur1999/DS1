void login()
{

    printf("\n\t\t\t\t\t User name: ");

    printf("\n\t\t\t\t\t Password: ");

        {

    fclose(log);
}

void AddTextOutput (char ans[1000])
{
    FILE *fp = fopen("Output.txt", "a");
    if (fp == NULL)
    {
        printf("\n\t\t\t\t 404! FILE not found!!!\n");
    }
    else
    {
        fflush(stdin);
        fprintf(fp,"%s",ans);
        fclose(fp);
    }
}

void reader ()
{
    system("cls");
    FILE *fp = fopen("Output.txt", "r");

    if (fp == NULL)
    {
        printf("\n\t\t\t\t Error while opening the file.\n");
    }

    printf("\n\t\t\t\t History: \n\n");
    char ch;
    while((ch = fgetc(fp)) != EOF)

    fclose(fp);
}

struct Point
{
    int x, y;
};


double getDistance(struct Point a, struct Point b)
{
    double distance;
    distance = sqrt((a.x - b.x) * (a.x - b.x) + (a.y-b.y) *(a.y-b.y));
    return distance;
}
void distance()

    sprintf(ans,"\n\t\t\t\t Distance between a(%d,%d) and b(%d,%d): %.2lf Unit\n", a.x, a.y, b.x, b.y, getDistance(a,b));

}
